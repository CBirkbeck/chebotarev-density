\section{Chebotarev density theorem}

The general Galois case is reduced to the abelian case via the cyclic
subextension generated by a class representative. Source: Sharifi 7.2.2
Step~1 (p.~143).

\begin{lemma}\label{lem:count-primes-above-frob-sigma}
  \lean{Chebotarev.count_primes_above_with_frobenius_eq_sigma}
  \uses{def:frobenius-at, def:frobenius-class}
  Let $L/K$ be a finite Galois extension with $G=\Gal{L/K}$,
  $C\subseteq G$ a conjugacy class, $\sigma\in C$, and $\fp$ a nonzero
  prime of $\OK$ unramified in $L$ with $\sigma_\fp = C$. The number
  of primes $\fP$ of $\Ocirc_L$ above $\fp$ with $\Frob_\fP = \sigma$
  is exactly $\abs{G}/(f\cdot\abs{C})$, where $f = \mathrm{ord}(\sigma)$.
\end{lemma}

\begin{proof}
  The Galois group $G$ acts transitively on primes above $\fp$, with
  stabiliser $D_\fP$. Hence the orbit has size $\abs{G}/\abs{D_\fP}$. For
  unramified $\fp$, $D_\fP=\ang{\Frob_\fP}$ has order $f$, so there are
  $\abs{G}/f$ primes above $\fp$. Their Frobenius elements form a
  $G$-orbit in $C$ (under conjugation), which is all of $C$ by
  transitivity; this orbit has size $\abs{C}$, and within each orbit
  the count of representatives $\sigma$ is $\abs{G}/(f\cdot\abs{C})$
  (Sharifi p.~143).
\end{proof}

\begin{lemma}[Density lift through the fixed field]\label{lem:density-lift-fixedfield}
  \lean{Chebotarev.density_lift_through_fixedField}
  \uses{def:dirichlet-density, def:frobenius-class, lem:count-primes-above-frob-sigma, lem:prime-ideal-sum-log}
  Let $\sigma\in\Gal{L/K}$, $E = L^{\ang\sigma}$, and $\sigma_E\in\Gal{L/E}$
  the corresponding element. Given the abelian-case density over $E$ for
  the Frobenius-fibre of $\sigma_E$ (value $1/\abs{\Gal{L/E}}$), the
  $K$-density of the Frobenius class $[\sigma]$ is $\abs{[\sigma]}/\abs{G}$.
\end{lemma}

\begin{proof}
  \uses{lem:count-primes-above-frob-sigma, lem:prime-ideal-sum-log}
  For primes $P$ of $E$ above $\fp\in S = \setof{\fp}{\sigma_\fp = [\sigma]}$,
  exactly $\abs{G}/(f\abs{C})$ of the primes of $L$ above have Frobenius
  $\sigma$ (\cref{lem:count-primes-above-frob-sigma}); with $\sum_\fp
  N\fp^{-s} \sim \sum_P NP^{-s}$ (\cref{lem:prime-ideal-sum-log} over both
  $K$ and $E$), $\delta_K(S) = (f\abs{C}/\abs{G})\,\delta_E(T_\sigma) =
  (f\abs{C}/\abs{G})(1/f) = \abs{C}/\abs{G}$.
\end{proof}

\begin{theorem}[Chebotarev density theorem]\label{thm:chebotarev-density}
  \lean{Chebotarev.chebotarev_density}
  \uses{thm:chebotarev-abelian, lem:density-lift-fixedfield}
  For a finite Galois extension $L/K$ of number fields with
  $G=\Gal{L/K}$ and any conjugacy class $C\subseteq G$,
  \[
    \delta\bigl(\setof{\fp\subset\OK}{\sigma_\fp = C}\bigr)
    \;=\; \frac{\abs{C}}{\abs{G}}.
  \]
\end{theorem}

\begin{proof}
  \uses{thm:chebotarev-abelian, lem:density-lift-fixedfield}
  \leanok
  Pick a representative $\sigma\in C$ (\texttt{ConjClasses.mk\_surjective})
  and let $E = L^{\ang\sigma}$ be the fixed field of $\ang\sigma$. The map
  $\ang\sigma \xrightarrow{\sim} \Gal{L/E}$
  (\texttt{IntermediateField.subgroupEquivAlgEquiv}) shows $\Gal{L/E}$ is
  commutative (as $\ang\sigma$ is cyclic), so $L/E$ is abelian and
  \cref{thm:chebotarev-abelian} applies to its Frobenius $\sigma_E$, giving
  $\delta_E(T_\sigma) = 1/\abs{\Gal{L/E}}$. Feeding this into the density
  lift \cref{lem:density-lift-fixedfield} yields $\abs{C}/\abs{G}$.
\end{proof}

\begin{theorem}[Chebotarev, abelian case]\label{thm:chebotarev-density-comm}
  \lean{Chebotarev.chebotarev_density_of_comm}
  \uses{thm:chebotarev-abelian, lem:conjclasses-carrier-card-one-comm}
  When $G = \Gal{L/K}$ is abelian, the Chebotarev density theorem holds:
  for every conjugacy class $C$,
  $\delta(\setof{\fp}{\sigma_\fp = C}) = \abs{C}/\abs{G}$.
\end{theorem}

\begin{proof}
  \uses{thm:chebotarev-abelian, lem:conjclasses-carrier-card-one-comm}
  \leanok
  In an abelian group each class $C = [\sigma]$ is a singleton
  (\cref{lem:conjclasses-carrier-card-one-comm}), so $\abs{C} = 1$ and
  the target density is $1/\abs{G}$; this is exactly
  \cref{thm:chebotarev-abelian} for the representative $\sigma$.
\end{proof}

\subsection*{Corollaries}

The corollaries below are derived directly by composition from
\cref{thm:chebotarev-density}. The Lean derivations use four small
helper sub-lemmas captured here.

\begin{lemma}\label{lem:conjclasses-mk-one-carrier-card-one}
  \lean{Chebotarev.ConjClasses_mk_one_carrier_card_eq_one}
  \leanok
  Let $G$ be a finite group. The conjugacy class of the identity has
  cardinality $1$:
  $\abs{(\mathrm{Conj}(G))_{\{1\}}} = 1$.
\end{lemma}

\begin{proof}\leanok
  The carrier is $\{1\}$ (by \texttt{isConj\_one\_left}, $a$ conjugate to
  $1$ iff $a = 1$); a singleton has cardinality $1$.
\end{proof}

\begin{lemma}\label{lem:conjclasses-carrier-card-one-comm}
  \lean{Chebotarev.ConjClasses_carrier_card_eq_one_of_comm}
  \leanok
  In a commutative finite group, every conjugacy class is a singleton,
  so has cardinality $1$.
\end{lemma}

\begin{proof}\leanok
  By \texttt{isConj\_iff\_eq} (conjugacy is equality in a commutative
  monoid), the carrier of $[g]$ is $\{g\}$.
\end{proof}

\begin{lemma}\label{lem:nat-card-galois-eq-finrank}
  \lean{Chebotarev.nat_card_galois_eq_finrank}
  \leanok
  For a finite-dimensional Galois extension $L/K$,
  $\abs{\Gal{L/K}} = [L:K]$.
\end{lemma}

\begin{proof}\leanok
  Mathlib's \texttt{IsGalois.card\_aut\_eq\_finrank}.
\end{proof}

\begin{lemma}\label{lem:infinite-of-density-pos}
  \lean{Chebotarev.infinite_of_hasDirichletDensity_pos}
  A set $S$ of prime ideals with positive Dirichlet density is infinite.
\end{lemma}

\begin{proof}
  Contrapositive: a finite set of primes has density $0$ by
  \cref{lem:has-density-finite} (the finite case of
  \cref{def:dirichlet-density}).
\end{proof}

\begin{lemma}\label{lem:conjclasses-carrier-card-pos}
  \lean{Chebotarev.ConjClasses_carrier_card_pos}
  \leanok
  Every conjugacy class in a finite group has cardinality at least $1$
  (it contains at least its own representative).
\end{lemma}

\begin{proof}\leanok
  A representative $a$ with $[a] = C$ lies in the carrier
  (\texttt{ConjClasses.mk\_surjective} + \texttt{mem\_carrier\_iff\_mk\_eq}),
  so the carrier is nonempty; a nonempty finite set has positive
  cardinality.
\end{proof}

\begin{corollary}\label{cor:density-split-completely}
  \lean{Chebotarev.density_split_completely}
  \uses{thm:chebotarev-density, lem:conjclasses-mk-one-carrier-card-one, lem:nat-card-galois-eq-finrank}
  For a finite Galois extension $L/K$ of number fields, the Dirichlet
  density of primes of $\OK$ that split completely in $L$ equals
  $1/[L:K]$.
\end{corollary}

\begin{proof}
  \uses{thm:chebotarev-density, lem:conjclasses-mk-one-carrier-card-one, lem:nat-card-galois-eq-finrank}
  A prime $\fp$ splits completely in $L$ iff every $\fP \mid \fp$ has
  $\Frob_\fP = 1$, i.e.\ the Frobenius conjugacy class is the trivial
  class. Apply \cref{thm:chebotarev-density} at $C = \mathrm{Conj}(G)_{\{1\}}$;
  by \cref{lem:conjclasses-mk-one-carrier-card-one} the numerator
  $\abs{C} = 1$ and by \cref{lem:nat-card-galois-eq-finrank} the
  denominator $\abs{G} = [L:K]$, so the density equals $1/[L:K]$.
\end{proof}

\begin{corollary}\label{cor:infinite-frobenius-class}
  \lean{Chebotarev.infinite_setOf_frobenius_class}
  \uses{thm:chebotarev-density, lem:infinite-of-density-pos, lem:conjclasses-carrier-card-pos}
  For a finite Galois extension $L/K$ of number fields and any conjugacy
  class $C \subseteq \Gal{L/K}$, the set of primes $\fp$ of $\OK$
  unramified in $L$ with Frobenius conjugacy class $C$ is infinite.
\end{corollary}

\begin{proof}
  \uses{thm:chebotarev-density, lem:infinite-of-density-pos, lem:conjclasses-carrier-card-pos}
  \cref{thm:chebotarev-density} gives Dirichlet density
  $\abs{C}/\abs{G}$, which is positive since $\abs{C} \ge 1$
  (\cref{lem:conjclasses-carrier-card-pos}) and $\abs{G} \ge 1$. Apply
  \cref{lem:infinite-of-density-pos}.
\end{proof}

\begin{corollary}[Dirichlet, density form]\label{cor:dirichlet-ap-density}
  \lean{Chebotarev.dirichlet_primes_in_AP}
  \uses{thm:chebotarev-density}
  For $n \ge 1$ and $a \in (\Z/n\Z)^\times$, the Dirichlet density of
  the rational primes $p$ with $p \equiv a \pmod{n}$ equals
  $1/\varphi(n)$.
\end{corollary}

\begin{proof}
  Specialise \cref{thm:chebotarev-density} to $K = \Q$, $L = \Q(\zeta_n)$.
  The cyclotomic character identifies $\Gal{\Q(\zeta_n)/\Q}$ with
  $(\Z/n\Z)^\times$; under this identification, the Frobenius
  $\sigma_p$ at unramified $p$ corresponds to $p \bmod n$. The Galois
  group is abelian, so each conjugacy class is a singleton; the set of
  primes with $p \equiv a \pmod{n}$ is the Frobenius preimage of
  $\{a\}$, whose density is $1/\varphi(n)$.
\end{proof}
