\section{Decomposition group, inertia, Frobenius}

Let $L/K$ be a finite Galois extension of number fields with
$G=\Gal{L/K}$. The Galois group acts on ideals of $\Ocirc_L$ (via the
canonical $G$-action restricted to $\Ocirc_L$ and extended
multiplicatively), and in particular permutes the primes lying above
a given prime $\fp$ of $\OK$.

\begin{definition}\label{def:unramified-in}
  \lean{Chebotarev.UnramifiedIn}
  \leanok
  A nonzero prime ideal $\fp$ of $\OK$ is \emph{unramified in $L$} if
  every prime $\fP$ of $\Ocirc_L$ lying over $\fp$ has ramification
  index $1$, i.e.\ $\fp\Ocirc_L$ factors into distinct primes.
\end{definition}

\begin{definition}\label{def:decomposition-group}
  \lean{Chebotarev.decompositionGroup}
  \leanok
  For a prime $\fP$ of $\Ocirc_L$, the \emph{decomposition group} at
  $\fP$ is
  \[
    D_\fP \;=\; \set{\sigma\in G : \sigma(\fP)=\fP}
    \;=\; \mathrm{Stab}_G(\fP).
  \]
\end{definition}

\begin{definition}\label{def:inertia-group}
  \lean{Chebotarev.inertiaGroup}
  \leanok
  \uses{def:decomposition-group}
  The \emph{inertia group} at $\fP$ is
  \[
    I_\fP \;=\;
    \setof{\sigma\in G}{\forall x\in\Ocirc_L,\;\sigma(x)\equiv x\pmod\fP}.
  \]
  This is the generic ideal-inertia subgroup \texttt{Ideal.inertia}
  from mathlib, instantiated at the natural action of $G$ on $\Ocirc_L$;
  acting trivially mod~$\fP$ forces stabilising $\fP$, so $I_\fP$ sits
  inside $D_\fP$, and is the kernel of the action of $D_\fP$ on the
  residue field $\Ocirc_L/\fP$.
\end{definition}

\begin{lemma}\label{lem:inertia-trivial-unramified}
  \lean{Chebotarev.inertiaGroup_trivial_of_unramified}
  \uses{def:inertia-group, def:unramified-in}
  If $\fP$ is unramified over $\fp=\fP\cap\OK$, then $I_\fP=\{1\}$.
\end{lemma}

\begin{proof}
  Mathlib's
  \texttt{Ideal.ramificationIdx\_eq\_one\_of\_isUnramifiedAt} gives the
  forward equivalence between unramifiedness and ramification index
  $1$; combined with $\abs{I_\fP}=e(\fP\mid\fp)$ this yields the
  claim.
\end{proof}

When $\fP$ is unramified, the surjection $D_\fP\to\Gal{(\Ocirc_L/\fP)/(\OK/\fp)}$
(mathlib's \texttt{IsFractionRing.stabilizerHom\_surjective}) has
trivial kernel \cref{lem:inertia-trivial-unramified}, so it is an
isomorphism. The residue Galois group of a finite-field extension is
canonically cyclic, generated by the absolute Frobenius
$x\mapsto x^{N\fp}$ (mathlib's
\texttt{FiniteField.frobeniusAlgEquivOfAlgebraic}). Hence $D_\fP$ has a
distinguished element.

\begin{theorem}\label{thm:exists-unique-frobenius}
  \lean{Chebotarev.exists_unique_frobeniusAt}
  \uses{def:decomposition-group, lem:inertia-trivial-unramified}
  Let $\fP$ be a nonzero prime of $\Ocirc_L$ lying over the unramified
  $\fp$. There is a unique $\sigma\in D_\fP$ such that
  \[
    \sigma(x)\equiv x^{N\fp}\pmod\fP \qquad\text{for every }x\in\Ocirc_L.
  \]
\end{theorem}

\begin{definition}\label{def:frobenius-at}
  \lean{Chebotarev.frobeniusAt}
  \uses{thm:exists-unique-frobenius}
  For an unramified nonzero prime $\fP$ of $\Ocirc_L$, the
  \emph{Frobenius automorphism} $\Frob_\fP\in G$ is the unique element
  of $D_\fP$ characterised by $\Frob_\fP(x)\equiv x^{N\fp}\pmod\fP$ for
  all $x\in\Ocirc_L$.
\end{definition}

\begin{definition}\label{def:frobenius-class}
  \lean{Chebotarev.frobeniusClass}
  \uses{def:frobenius-at}
  For a nonzero prime $\fp$ of $\OK$ unramified in $L$, the
  \emph{Frobenius conjugacy class} is
  \[
    \sigma_\fp \;=\; \set{\Frob_\fP : \fP\mid\fp}
    \;\in\; \mathrm{Conj}(G),
  \]
  the conjugacy class in $G$ containing $\Frob_\fP$ for any (equivalently
  every) prime $\fP$ of $\Ocirc_L$ above $\fp$. The class is independent
  of the choice of $\fP$.
\end{definition}

\begin{lemma}\label{lem:exists-frobenius-class}
  \lean{Chebotarev.exists_frobeniusClass}
  \uses{def:frobenius-at}
  For a nonzero prime $\fp$ of $\OK$ unramified in $L$, there exists a
  conjugacy class $C \subseteq \Gal{L/K}$ such that for every prime
  $\fP$ of $\Ocirc_L$ above $\fp$, $C = [\Frob_\fP]$. (Equivalently, the
  Frobenius elements above $\fp$ are all conjugate.)
\end{lemma}

\begin{lemma}\label{lem:frobenius-class-eq-mk}
  \lean{Chebotarev.frobeniusClass_eq_mk_frobeniusAt}
  \uses{def:frobenius-class, def:frobenius-at, lem:exists-frobenius-class}
  For a nonzero prime $\fp$ of $\OK$ unramified in $L$, and any prime
  $\fP$ of $\Ocirc_L$ above $\fp$,
  \[
    \sigma_\fp \;=\; [\Frob_\fP].
  \]
\end{lemma}

\begin{proof}
  \uses{lem:exists-frobenius-class}
  \leanok
  Unfold $\sigma_\fp$ at the unramified positive case to expose the
  $\mathrm{choose}$ from \cref{lem:exists-frobenius-class}; the witness
  property of $\mathrm{choose\_spec}$ at $\fP$ closes the goal.
\end{proof}

\begin{lemma}\label{lem:finite-ramified-primes}
  \lean{Chebotarev.finite_ramifiedIn}
  \uses{def:unramified-in}
  Only finitely many nonzero primes of $\OK$ ramify in $L$.
\end{lemma}

\begin{proof}
  The primes ramifying in $L$ are exactly the divisors of the relative
  different ideal $\mathfrak{d}_{L/K}\subseteq\Ocirc_L$, which has a
  finite prime factorisation in the Dedekind domain $\Ocirc_L$.
\end{proof}
